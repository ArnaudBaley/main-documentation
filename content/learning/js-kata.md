#### Even Or Odd :
```js
function even_or_odd(number) {
  if (number % 2 === 0) return 'Even';
  return 'Odd';
}

console.log(even_or_odd(4)) // Even
console.log(even_or_odd(5)) // Odd
```

#### Opposite number :
```js
function opposite(number) {
   return -number
}

console.log(opposite(5)) // -5
```
